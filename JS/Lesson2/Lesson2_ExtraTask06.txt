Есть два слова: «Welcome» и «Вэлкам», каждая четвертая буква которых имеет свой 
вес. Определите, какая из них весит больше, во сколько раз (и удивитесь, насколько 
отличается вес английских и русских букв :). А затем проверьте бо́льшую букву на 
чётность/нечётность.


 let str1 = 'Welcome';
 let str2 = 'Вэлкам';
 let l1 = str1.charCodeAt(3);
 let l2 = str2.charCodeAt(3);
 console.log(l1, l2);
 console.log(l1 > l2);
 console.log(l2 / l1);
 console.log(l2 % 2 == 0);
