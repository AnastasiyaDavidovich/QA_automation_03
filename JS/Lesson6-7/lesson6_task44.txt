Given a number n, draw stairs using the letter "I", n tall and n wide, with the tallest in the 
top left. For example n = 3 result in: "I\n I\n  I"


 function drawStairs(n) {
 let a = 'I';
   for (let i=1; i<n; i++){
    a += '\n' + ' '.repeat(i)+'I';
  }
   return a;
 }
